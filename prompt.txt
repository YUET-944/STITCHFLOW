CREATE OR REPLACE FUNCTION vault_set_current() RETURNS TRIGGER AS $$
BEGIN
  UPDATE measurements SET is_current = false
  WHERE client_id = NEW.client_id AND id != NEW.id;
  RETURN NEW;
END; $$ LANGUAGE plpgsql;

  AFTER INSERT ON measurements
  FOR EACH ROW EXECUTE FUNCTION vault_set_current();
2.1.2  Triple-Gate Security — MANDATORY
⚠ CRITICAL
ALL THREE GATES MUST PASS for any write to the Measurement Vault.
A compromised API layer cannot bypass Gate 3 (RLS). This is non-negotiable.
Implement in this exact order: JWT Role → Link Existence → RLS Policy.
  Triple-Gate Security — Full Implementation
// ── GATE 1: JWT Role Check (Fastify Middleware) ──────────────────
// Applied to ALL /measurements routes via preHandler hook
async function tailorRoleGuard(request, reply) {
  const { role } = request.user; // decoded JWT payload
  if (role !== 'TAILOR') {
    return reply.code(403).send({
      code: 'ROLE_FORBIDDEN',
      message: 'Only TAILOR role may write to the Vault'
    });
  }
}
// ── GATE 2: Link Existence Check (Service Layer) ─────────────────
// Executed INSIDE the measurements service before any DB write
async function assertActiveLink(tailorId, clientId, prisma) {
  const link = await prisma.tailorClientLink.findFirst({
    where: { tailor_id: tailorId, client_id: clientId, is_active: true }
  });
  if (!link) throw new ForbiddenError({
    code: 'LINK_NOT_FOUND',
    message: 'No active order link between tailor and client'
  });
}
// ── GATE 3: Supabase RLS Policy (Database Layer) ─────────────────
-- Applied directly in PostgreSQL; cannot be bypassed by API layer
ALTER TABLE measurements ENABLE ROW LEVEL SECURITY;
CREATE POLICY measurement_write ON measurements
  FOR INSERT WITH CHECK (
    tailor_id = auth.uid()
    AND EXISTS (
      SELECT 1 FROM tailor_client_links
      WHERE tailor_id = auth.uid()
        AND client_id = measurements.client_id
        AND is_active = true
    )
  );
-- Read policy: clients read their own; tailors read linked clients only
CREATE POLICY measurement_read ON measurements FOR SELECT USING (
  client_id = auth.uid()
  OR (
    tailor_id = auth.uid() AND EXISTS (
      SELECT 1 FROM tailor_client_links
      WHERE tailor_id = auth.uid()
        AND client_id = measurements.client_id
        AND is_active = true
    )
  )
);
2.2  Capacity Gate — Serialised Transaction (Race-Condition Proof)
Tailors configure a Max Active Orders ceiling. The system MUST enforce this ceiling atomically. Without serialisation, two simultaneous booking requests can both pass the capacity check and both create orders, exceeding the ceiling. This is a hard correctness requirement.
⚠ CRITICAL
Use SERIALIZABLE isolation + SELECT ... FOR UPDATE on tailor_profiles row.
Never use a simple SELECT followed by INSERT — this creates a TOCTOU race condition.
The FOR UPDATE lock ensures only one transaction holds the row lock at a time.
  Capacity Gate — Serialised Transaction (bookingService.ts)
// bookingService.ts — createBooking()
export async function createBookingWithCapacityGate(
  tailorId: string,
  bookingPayload: CreateBookingDTO,
  prisma: PrismaClient
): Promise<Order> {
  return await prisma.$transaction(async (tx) => {
    // ── Step 1: Acquire row-level lock on tailor profile ─────
    // FOR UPDATE prevents any other transaction from reading
    // or modifying this row until this transaction commits/rolls back
    const tailor = await tx.$queryRaw<TailorProfile[]>`
      SELECT current_active_orders, max_active_orders, availability_status
      FROM tailor_profiles
      WHERE tailor_id = ${tailorId}
      FOR UPDATE
    `;
    if (!tailor.length) throw new NotFoundError({ code: 'TAILOR_NOT_FOUND' });
    const t = tailor[0];
    // ── Step 2: Capacity enforcement gate ────────────────────
    if (t.current_active_orders >= t.max_active_orders) {
      throw new CapacityExceededException({
        code:    'CAPACITY_EXCEEDED',
        http:    409,
        message: `Tailor at full capacity (${t.current_active_orders}/${t.max_active_orders})`,
      });
    }
    // ── Step 3: Availability status gate ─────────────────────
    if (t.availability_status !== 'ACTIVE') {
      throw new UnavailableException({
        code:    'TAILOR_UNAVAILABLE',
        http:    409,
        message: 'Tailor is not currently accepting bookings',
      });
    }
    // ── Step 4: Create order (within the same locked transaction)
    const order = await tx.order.create({ data: { ...bookingPayload, tailor_id: tailorId } });
    return order;
    // COMMIT: lock released; capacity enforced atomically
  }, { isolationLevel: Prisma.TransactionIsolationLevel.Serializable });
}

// posService.ts — executePOSAction()
export async function executePOSAction(orderId: string, posPayload: POSPayload) {
  // ── Phase 1: Persist financial data (within DB transaction) ──
  const { invoice, order, garments } = await prisma.$transaction(async (tx) => {
    const invoice = await tx.invoice.create({
      data: {
        id:             generateInvoiceId(posPayload.tailorShort),
        order_id:       orderId,
        tailor_id:      posPayload.tailorId,
        client_id:      posPayload.clientId,
        base_price:     posPayload.basePrice,
        advance_paid:   posPayload.advancePaid,
        balance_due:    posPayload.basePrice - posPayload.advancePaid,
        payment_status: posPayload.advancePaid > 0 ? 'ADVANCE_RECEIVED' : 'BALANCE_PENDING',
      }
    });
    const updatedOrder = await tx.order.update({
      where: { id: orderId },
      data:  { invoice_id: invoice.id, booking_status: 'CONFIRMED' }
    });
    const garments = await tx.orderGarment.findMany({ where: { order_id: orderId } });
    return { invoice, order: updatedOrder, garments };
  });
  // ── Phase 2: Generate both documents in parallel ──────────
  const [invoicePdfUrl, stitchingTicketUrl] = await Promise.all([
    generateInvoicePDF(invoice, order),       // PDFKit → Cloudinary
    generateStitchingTicket(order, garments), // PDFKit + QR → Cloudinary
  ]);
  // ── Phase 3: Persist PDF URLs back to records ─────────────
  await Promise.all([
    prisma.invoice.update({ where:{id:invoice.id}, data:{pdf_url:invoicePdfUrl} }),
    prisma.stitchingTicket.update({ where:{order_id:orderId}, data:{pdf_url:stitchingTicketUrl} }),
  ]);
  return { invoice, invoicePdfUrl, stitchingTicketUrl };
}

// POST /api/v1/auth/register
// Body: { phone, role: 'TAILOR' | 'CLIENT', firebaseIdToken, ...profileFields }
async function register(body: RegisterDTO) {
  // Step 1: Verify Firebase ID token (phone OTP already validated by Firebase client SDK)
  const firebaseUser = await admin.auth().verifyIdToken(body.firebaseIdToken);
  if (firebaseUser.phone_number !== normalizeE164(body.phone))
    throw new UnauthorizedError({ code: 'PHONE_MISMATCH' });
  // Step 2: Create user record
  const user = await prisma.user.create({
    data: {
      phone:       firebaseUser.phone_number,
      role:        body.role,  // 'TAILOR' or 'CLIENT' — immutable
      readable_id: await generateReadableId(body.role, body.city),
      firebase_uid: firebaseUser.uid,
      ...roleSpecificFields(body),
    }
  });
  // Step 3: Mint JWT with role claim
  const accessToken  = signJWT({ sub: user.id, role: user.role }, '60m');
  const refreshToken = signRefreshJWT({ sub: user.id }, '30d');
  return { user, accessToken, refreshToken };
}
